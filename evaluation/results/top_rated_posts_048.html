Our Solution


SELECT
    T1.CName,
    T0.AddressID,
    T0.AddressLine 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            CName,
            min(AddressID) AS min_AddressID 
        FROM
            input1 
        GROUP BY
            CName
    ) AS T1 
        ON T1.CName = T0.CName 
        AND T1.min_AddressID = T0.AddressID
