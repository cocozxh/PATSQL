Our Solution


SELECT
    T0.Section,
    T0.Count,
    T1.Count 
FROM
    input1 AS T0 
JOIN
    input1 AS T1 
        ON T0.Section = T1.Section 
WHERE
    T0.Status = 'Active' 
    AND T1.Status = 'Inactive' 
ORDER BY
    T0.Section ASC
