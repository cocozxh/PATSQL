Our Solution


SELECT
    T0.id,
    T1.user,
    T0.time,
    T0.io 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            user,
            max(time) AS max_time 
        FROM
            input1 
        GROUP BY
            user
    ) AS T1 
        ON T1.user = T0.user 
        AND T1.max_time = T0.time 
ORDER BY
    T0.time ASC
