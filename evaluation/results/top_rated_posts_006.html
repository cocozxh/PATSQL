Our Solution


SELECT
    T1.max_Id,
    T1.Name,
    T0.Other_Columns 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            Name,
            max(Id) AS max_Id 
        FROM
            input1 
        GROUP BY
            Name
    ) AS T1 
        ON T1.max_Id = T0.Id 
ORDER BY
    T1.Name ASC
