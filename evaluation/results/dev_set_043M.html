Our Solution


SELECT
    T0.Id,
    T0.name,
    T1.type,
    T0.price 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            type,
            max(price) AS max_price 
        FROM
            input1 
        GROUP BY
            type
    ) AS T1 
        ON T1.type = T0.type 
        AND T1.max_price = T0.price 
ORDER BY
    T1.type ASC
