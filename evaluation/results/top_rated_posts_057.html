Our Solution


SELECT
    T0.Name,
    T1.City,
    T0.Birthyear 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            City,
            min(Birthyear) AS min_Birthyear 
        FROM
            input1 
        GROUP BY
            City
    ) AS T1 
        ON T1.City = T0.City 
        AND T1.min_Birthyear = T0.Birthyear
