Our Solution


SELECT
    T1.MATERIAL,
    T0.DISCO_DATE,
    T1.max_DATE_UPDATE 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            MATERIAL,
            max(DATE_UPDATE) AS max_DATE_UPDATE 
        FROM
            input1 
        GROUP BY
            MATERIAL
    ) AS T1 
        ON T1.MATERIAL = T0.MATERIAL 
        AND T1.max_DATE_UPDATE = T0.DATE_UPDATE 
ORDER BY
    T1.MATERIAL ASC
