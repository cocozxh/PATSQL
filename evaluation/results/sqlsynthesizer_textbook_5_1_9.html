Our Solution


SELECT
    max(T1.fname),
    count(T0.ID_class) 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.ID_key = T1.ID_key 
GROUP BY
    T0.ID_key 
HAVING
    max(T0.room) = 'R128' 
    AND min(T0.room) = 'R128' 
ORDER BY
    max(T1.fname) ASC
