Our Solution


SELECT
    T1.id,
    T1.max_rev,
    T0.content 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            id,
            max(rev) AS max_rev 
        FROM
            input1 
        GROUP BY
            id
    ) AS T1 
        ON T1.id = T0.id 
        AND T1.max_rev = T0.rev 
ORDER BY
    T1.id ASC
