Our Solution


SELECT
    T1.ARIDNR,
    T0.LIEFNR 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            ARIDNR,
            count(ARIDNR) AS count_ARIDNR 
        FROM
            input1 
        GROUP BY
            ARIDNR
    ) AS T1 
        ON T1.ARIDNR = T0.ARIDNR 
WHERE
    T1.count_ARIDNR > 1 
ORDER BY
    T1.ARIDNR ASC
