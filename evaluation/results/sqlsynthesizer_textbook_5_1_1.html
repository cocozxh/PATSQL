Our Solution


SELECT
    T3.S_name 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.C_name = T1.C_name 
JOIN
    input3 AS T2 
        ON T0.F_key = T2.F_key 
JOIN
    input4 AS T3 
        ON T1.S_key = T3.S_key 
WHERE
    T3.level = 'JR' 
    AND T2.F_name = 'faculty1' 
ORDER BY
    T3.S_name ASC
