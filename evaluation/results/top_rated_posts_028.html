Our Solution


SELECT
    T0.PID,
    string_agg(T1.Descr, ' ') 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.PID = T1.PID 
GROUP BY
    T0.PID 
ORDER BY
    T0.PID ASC
