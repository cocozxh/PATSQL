Our Solution


SELECT
    T0.isbn,
    max(T0.title),
    string_agg(T1.Author, ', ') 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.isbn = T1.Isbn_id 
GROUP BY
    T0.isbn 
ORDER BY
    T0.isbn ASC
