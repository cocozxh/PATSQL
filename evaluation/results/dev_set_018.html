Our Solution


SELECT
    T1.Id,
    T0.Plan,
    T1.Attributes,
    T1.Value 
FROM
    input1 AS T0 
JOIN
    input1 AS T1 
        ON T0.Plan = T1.Plan 
WHERE
    T0.Value = 'True' 
    AND T0.Attributes = 'IsActive' 
ORDER BY
    T1.Id ASC
