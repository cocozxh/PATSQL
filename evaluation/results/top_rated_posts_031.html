Our Solution


SELECT
    T0.id,
    T0.rank_over_order_by_name_asc,
    T0.name 
FROM
    (SELECT
        id,
        name,
        rank() OVER (ORDER BY name ASC) AS rank_over_order_by_name_asc 
    FROM
        input1) AS T0 
WHERE
    T0.name = 'Beta'
