Our Solution


SELECT
    T0.NAME,
    T0.SCORE 
FROM
    (SELECT
        NAME,
        SCORE,
        rank() OVER (PARTITION BY NAME ORDER BY SCORE DESC) AS rank_over_part_by_name_order_by_score_desc 
    FROM
        input1) AS T0 
WHERE
    T0.rank_over_part_by_name_order_by_score_desc <= 2
