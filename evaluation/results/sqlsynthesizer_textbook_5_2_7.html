Our Solution


SELECT
    T0.supplier_key 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.part_id = T1.part_id 
GROUP BY
    T0.supplier_key 
HAVING
    max(T1.color) = 'red' 
    AND min(T1.color) = 'red' 
ORDER BY
    T0.supplier_key ASC
