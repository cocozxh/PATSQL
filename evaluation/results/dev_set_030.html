Our Solution


SELECT
    T0.ID,
    max(T0.DESCR),
    max(T1.LAST_DATE) 
FROM
    input1 AS T0 
LEFT JOIN
    input2 AS T1 
        ON T0.ID = T1.LINK 
GROUP BY
    T0.ID 
ORDER BY
    T0.ID ASC
