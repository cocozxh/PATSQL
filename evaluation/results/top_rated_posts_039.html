Our Solution


SELECT
    T2.c 
FROM
    (SELECT
        max(T0.count_c) AS max_T0_count_c 
    FROM
        (SELECT
            count(c) AS count_c 
        FROM
            input1 
        GROUP BY
            c) AS T0) AS T1 
    JOIN
        (
            SELECT
                c,
                count(c) AS count_c 
            FROM
                input1 
            GROUP BY
                c
        ) AS T2 
            ON T1.max_T0_count_c = T2.count_c
