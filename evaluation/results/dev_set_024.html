Our Solution


SELECT
    T1.user_id,
    T0.names 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            user_id,
            count(user_id) AS count_user_id 
        FROM
            input1 
        GROUP BY
            user_id
    ) AS T1 
        ON T1.user_id = T0.user_id 
WHERE
    T1.count_user_id > 3 
ORDER BY
    T0.names ASC
