Our Solution


SELECT
    T1.Member,
    count(T0.Member) 
FROM
    (SELECT
        Member,
        Element 
    FROM
        t1) AS T0 
LEFT JOIN
    (
        SELECT
            Member,
            Element 
        FROM
            t1
    ) AS T1 
        ON T0.Element = T1.Element 
WHERE
    T0.Member = '1' 
    AND T1.Member <> '1' 
GROUP BY
    T1.Member 
ORDER BY
    T1.Member ASC
