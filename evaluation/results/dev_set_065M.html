Our Solution


SELECT
    T0.id_orden,
    T0.date,
    T0.total,
    T1.name,
    T1.phone,
    T2.max_edo 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.id_usuario = T1.id_usuario 
JOIN
    (
        SELECT
            id_orden,
            max(edo) AS max_edo 
        FROM
            input3 
        GROUP BY
            id_orden
    ) AS T2 
        ON T0.id_orden = T2.id_orden 
ORDER BY
    T0.id_orden ASC
