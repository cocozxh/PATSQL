Our Solution


SELECT
    max(T0.Name),
    sum(T1.Quantity) 
FROM
    input1 AS T0 
LEFT JOIN
    input2 AS T1 
        ON T0.ID = T1.Menu_id 
GROUP BY
    T0.ID 
ORDER BY
    max(T0.Name) ASC
