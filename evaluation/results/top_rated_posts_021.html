Our Solution


SELECT
    T0.CHARGEID,
    T1.CHARGETYPE,
    T0.SERVICEMONTH 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            CHARGETYPE,
            max(SERVICEMONTH) AS max_SERVICEMONTH 
        FROM
            input1 
        GROUP BY
            CHARGETYPE
    ) AS T1 
        ON T1.CHARGETYPE = T0.CHARGETYPE 
        AND T1.max_SERVICEMONTH = T0.SERVICEMONTH 
ORDER BY
    T0.CHARGEID ASC
