Our Solution


SELECT
    indicator_code,
    indicator_name,
    count(country_name) 
FROM
    international_education 
WHERE
    year = 2016 
GROUP BY
    indicator_name,
    indicator_code 
HAVING
    count(country_name) >= 5 
ORDER BY
    indicator_name DESC
