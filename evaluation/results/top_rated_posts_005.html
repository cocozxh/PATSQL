Our Solution


SELECT
    T1.min_id,
    T1.customer,
    T0.total 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            customer,
            min(id) AS min_id 
        FROM
            input1 
        GROUP BY
            customer
    ) AS T1 
        ON T1.min_id = T0.id 
ORDER BY
    T1.customer ASC
