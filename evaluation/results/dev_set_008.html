Our Solution


SELECT
    T0.a,
    T1.max_b 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            max(b) AS max_b 
        FROM
            input1
    ) AS T1 
        ON T1.max_b = T0.b 
ORDER BY
    T0.a ASC
