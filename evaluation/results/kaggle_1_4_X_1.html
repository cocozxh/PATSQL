Our Solution


SELECT
    country_name,
    avg(value) 
FROM
    international_education 
WHERE
    indicator_code = 'SE.XPD.TOTL.GD.ZS' 
    AND year >= 2010 
    AND year <= 2017 
GROUP BY
    country_name 
ORDER BY
    country_name DESC
