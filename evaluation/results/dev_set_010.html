Our Solution


SELECT
    T0.id,
    T0.country,
    T0.status 
FROM
    input1 AS T0 
JOIN
    input1 AS T1 
        ON T0.country = T1.country 
WHERE
    T0.status = 'TREATED' 
    AND T1.status = 'UNTREATED' 
ORDER BY
    T0.id ASC
