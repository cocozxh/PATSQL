Our Solution


SELECT
    T1.Usr,
    T0.Phone,
    T1.max_Value 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            Usr,
            max(Value) AS max_Value 
        FROM
            input1 
        WHERE
            Value < 8 
        GROUP BY
            Usr
    ) AS T1 
        ON T1.Usr = T0.Usr 
        AND T1.max_Value = T0.Value 
ORDER BY
    T1.Usr DESC
