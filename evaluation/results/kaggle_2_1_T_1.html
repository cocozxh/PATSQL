Our Solution


SELECT
    T0.parent,
    max(T1.by),
    max(T1.title),
    count(T0.id) 
FROM
    comments AS T0 
JOIN
    stories AS T1 
        ON T0.parent = T1.id 
WHERE
    T0.time_ts >= DATE '2012-01-01' 
GROUP BY
    T0.parent 
ORDER BY
    count(T0.id) DESC
