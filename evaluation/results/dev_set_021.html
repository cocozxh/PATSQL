Our Solution


SELECT
    T0.Customer,
    T1.email,
    T0.ZIP,
    T0.shop 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            email,
            count(Customer) AS count_Customer 
        FROM
            input1 
        GROUP BY
            email
    ) AS T1 
        ON T1.email = T0.email 
WHERE
    T1.count_Customer > 1 
ORDER BY
    T0.shop ASC,
    T0.Customer DESC
