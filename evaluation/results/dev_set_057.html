Our Solution


SELECT
    T0.CODE 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            max(Name) AS max_Name,
            max(QTY) AS max_QTY 
        FROM
            input1 
        WHERE
            Name = 'Bottle'
    ) AS T1 
        ON T1.max_Name = T0.Name 
        AND T1.max_QTY = T0.QTY
