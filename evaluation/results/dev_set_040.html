Our Solution


SELECT
    T0.id,
    T0.date,
    T0.phone_number 
FROM
    call AS T0 
LEFT JOIN
    phone_book AS T1 
        ON T0.phone_number = T1.phone_number 
WHERE
    T1.id IS NULL 
ORDER BY
    T0.id ASC
