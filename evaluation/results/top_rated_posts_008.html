Our Solution


SELECT
    max(T0.OrderNumber),
    max(T1.Quantity),
    max(T1.Description) 
FROM
    input1 AS T0 
JOIN
    input2 AS T1 
        ON T0.OrderID = T1.OrderID 
GROUP BY
    T0.OrderID 
ORDER BY
    max(T0.OrderNumber) DESC
