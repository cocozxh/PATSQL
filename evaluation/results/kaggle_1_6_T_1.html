Our Solution


SELECT
    T1.license,
    count(T0.repo_name) 
FROM
    sample_files AS T0 
JOIN
    licenses AS T1 
        ON T0.repo_name = T1.repo_name 
GROUP BY
    T1.license 
ORDER BY
    count(T0.repo_name) DESC
