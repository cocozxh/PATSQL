Our Solution


SELECT
    T0.ID,
    T0.Name 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            Parent 
        FROM
            input1 
        GROUP BY
            Parent
    ) AS T1 
        ON T1.Parent = T0.ID 
ORDER BY
    T0.ID ASC
