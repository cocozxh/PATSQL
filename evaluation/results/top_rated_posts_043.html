Our Solution


SELECT
    T0.CustomerName,
    T1.EmailAddress 
FROM
    input1 AS T0 
JOIN
    (
        SELECT
            EmailAddress,
            count(CustomerName) AS count_CustomerName 
        FROM
            input1 
        GROUP BY
            EmailAddress
    ) AS T1 
        ON T1.EmailAddress = T0.EmailAddress 
WHERE
    T1.count_CustomerName > 1 
ORDER BY
    T1.EmailAddress ASC
