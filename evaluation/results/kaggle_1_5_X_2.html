Our Solution


SELECT
    EXTRACT(month FROM trip_start_timestamp),
    count(unique_key) 
FROM
    taxi_trips 
WHERE
    EXTRACT(year FROM trip_start_timestamp) = 2017 
GROUP BY
    EXTRACT(month FROM trip_start_timestamp) 
ORDER BY
    EXTRACT(month FROM trip_start_timestamp) ASC
